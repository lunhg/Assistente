const __index__ = function(name, fn){
  let index_path = path.join(app.get('assets'), name);
  fs.readFile(index_path, function(err, data){
    if(err) {
      logger.debug(err);
      fn(err);
    } else {
      logger.debug('Sending '+index_path);
      fn(data);
    }
  });
};
